Lệnh Triển Khai (5 Bước)
# ─── BƯỚC 1: Môi Trường Node.js (arm64) ───
curl -o- https://raw.githubusercontent.com/nvm-sh/nvm/v0.39.7/install.sh | bash
nvm install 24 && nvm use 24 && nvm alias default 24
npm install -g pnpm pm2

# ─── BƯỚC 2: Cài OpenClaw ───
curl -fsSL https://openclaw.ai/install.sh | bash
openclaw onboard # Wizard cấu hình Telegram + allowFrom whitelist

# ─── BƯỚC 3: Free Ride Engine ───
npx clawhub@latest install free-ride
echo 'export OPENROUTER_API_KEY="sk-or-v1-YOUR_KEY"' >> ~/.zshrc && source ~/.zshrc

# ─── BƯỚC 4: Affiliate Skills ───
git clone https://github.com/Affitor/affiliate-skills.git ~/.openclaw/skills/affiliate-skills
npx clawhub@latest install mcporter

# ─── BƯỚC 5: PM2 24/7 ───
pm2 start openclaw --name "OpenClaw-Gateway" -- start
pm2 startup && pm2 save
# ✓ Gateway đang chạy nền 24/7 — không bao giờ tắt
